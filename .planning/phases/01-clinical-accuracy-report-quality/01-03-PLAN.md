    ```typescript
    'use client';

    import { getStandards } from '@/lib/normative/ratings';
    import type { RatingTier, TierRanges } from '@/types/normative';

    interface RangeBarProps {
      value: number;
      testKey: string;

      gender?: string | null;
    }

    const SEGMENT_COLORS: Record<RatingTier, string> = {
      poor: 'bg-red-500',
      cautious: 'bg-amber-400',
      normal: 'bg-gray-400',
      great: 'bg-blue-500',
      elite: 'bg-emerald-500',
    };

    const TIERS: RatingTier[] = ['poor', 'cautious', 'normal', 'great', 'elite'];
    const MIN_SEGMENT_PCT = 10; // Minimum 10% width per segment (Pitfall 2)
    ```

    Component logic:
    - Call `getStandards(testKey, age, gender)` to get the TierRanges
    - If no standards, return null
    - Calculate each tier's proportional width: `tierWidth = (tier.max - tier.min) / totalRange * 100`
    - Apply minimum width enforcement: if any segment < MIN_SEGMENT_PCT, set to MIN_SEGMENT_PCT and proportionally reduce others (redistribute the borrowed percentage from the widest segments)
    - Calculate needle position: `needlePercent = ((clampedValue - fullMin) / totalRange) * 100`
    - Clamp value to fullMin..fullMax range
    - Render: outer `div` with `className="relative w-full h-3 flex rounded-full overflow-hidden"`, inner divs for each tier segment with `style={{ width: '${tierWidths[i]}%' }}` and SEGMENT_COLORS class, and a needle div `className="absolute top-0 h-full w-0.5 bg-[#1a365d]"` at `left: ${needlePercent}%` with a small circle on top
