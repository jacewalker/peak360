  <action>
    Create `src/lib/markers/__tests__/integration.test.ts`. This test exercises the real query layer against the real DB (using whatever DATABASE_URL is set in the test env - SQLite is fine; the test must NOT require Postgres specifically). Pattern:

    ```typescript
    import { describe, it, expect, beforeAll, afterAll } from 'vitest';
    import { createMarker, deleteMarker, getAllMarkers } from '@/lib/markers/queries';
    import { getReportMarkers } from '@/lib/markers/registry';

    describe('Phase 12 integration: marker registry end-to-end', () => {
      const testKey = `phase12_integration_${Date.now()}`;
      let created: Awaited<ReturnType<typeof createMarker>>;

      beforeAll(async () => {
        created = await createMarker({
          testKey,
          label: 'Phase 12 Integration Marker',
          section: 5,
          dataKey: 'phase12IntegrationMarker',
          pillar: 'cardiometabolic',
          category: 'lipids',
          subcategory: null,
          fallbackUnit: 'mg/dL',
          hasNorms: false,
          aiAliases: ['phase 12 integration'],
          severityWeight: null,
          createdBy: 'integration-test',
          updatedBy: 'integration-test',
        });
      });

      afterAll(async () => {
        await deleteMarker(testKey);
      });

      it('returns the created marker from getAllMarkers', async () => {
        const all = await getAllMarkers();
        expect(all.some((m) => m.testKey === testKey)).toBe(true);
      });

      it('merges into getReportMarkers with source=db and the stored pillar', async () => {
        const merged = await getReportMarkers();
        const row = merged.find((m) => m.testKey === testKey);
        expect(row).toBeDefined();
        expect(row?.source).toBe('db');
        expect(row?.pillar).toBe('cardiometabolic');
      });

      it('markerToPillar short-circuits to the DB-stored pillar', async () => {
        const merged = await getReportMarkers();
        const row = merged.find((m) => m.testKey === testKey);
        const { markerToPillar } = await import('@/lib/pillars/mapping');
        const result = markerToPillar(row as unknown as Parameters<typeof markerToPillar>[0]);
        expect(result.pillar).toBe('cardiometabolic');
      });
    });
    ```

    If the test environment cannot reach a writable DB (e.g. CI with no DATABASE_URL and no local SQLite), gate the test with `describe.skipIf(...)` or document the requirement in the test file's header comment.
