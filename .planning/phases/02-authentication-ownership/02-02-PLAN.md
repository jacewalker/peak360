   - Remove import of `validateSessionToken` from `@/lib/auth/session` (deleted in Plan 01)
   - Import `getSessionCookie` from `better-auth/cookies`
   - Keep the PORTAL_SUBDOMAIN_HOSTNAMES redirect logic unchanged
   - Keep static asset passthrough unchanged
   - Update PUBLIC_PATHS to: `/login`, `/api/auth/*` (Better Auth catch-all needs to be public), `/api/health`
   - For portal paths (`/portal/*`): check `getSessionCookie(req)` instead of the old HMAC validation. If no cookie, redirect to `/login` (for pages) or return 401 JSON (for API).
   - For admin paths (`/portal/admin/*`): optimistic cookie check only (real validation in API handlers). Just ensure the session cookie exists.
   - Remove all references to `ADMIN_PASSWORD` env var
   - Keep the `config.matcher` unchanged
   - Landing page paths (non-portal, non-api) remain public (no auth required)

   Key structure:
   ```typescript
   import { NextRequest, NextResponse } from 'next/server';
   import { getSessionCookie } from 'better-auth/cookies';

   const PUBLIC_PATHS = new Set(['/login', '/api/health']);

   const PORTAL_SUBDOMAIN_HOSTNAMES = new Set(
     (process.env.PORTAL_HOSTNAMES ?? 'portal.peak360.com.au').split(',')
   );

   export async function middleware(req: NextRequest) {
     const { pathname } = req.nextUrl;

     // Static assets - pass through
     if (pathname.startsWith('/_next') || pathname.startsWith('/favicon') ||
         pathname.endsWith('.png') || pathname.endsWith('.svg') || pathname.endsWith('.ico')) {
       return NextResponse.next();
     }

     // Portal subdomain redirect (preserved from existing)
     const hostname = req.headers.get('host')?.split(':')[0] ?? '';
     if (PORTAL_SUBDOMAIN_HOSTNAMES.has(hostname)) {
       const targetPath = pathname === '/' ? '/portal' : `/portal${pathname}`;
       const targetUrl = new URL(targetPath, 'https://peak360.com.au');
       return NextResponse.redirect(targetUrl);
     }

     // Public paths + Better Auth routes
     if (PUBLIC_PATHS.has(pathname) || pathname.startsWith('/api/auth/')) {
       return NextResponse.next();
     }

     // Landing page (non-portal, non-api) - public
     if (!pathname.startsWith('/portal') && !pathname.startsWith('/api/')) {
       return NextResponse.next();
     }

     // Protected paths: check session cookie (optimistic, NOT security boundary)
     const sessionCookie = getSessionCookie(req);
     if (!sessionCookie) {
       if (pathname.startsWith('/api/')) {
         return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
       }
       return NextResponse.redirect(new URL('/login', req.url));
     }

     return NextResponse.next();
   }
   ```
